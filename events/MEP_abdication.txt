		if = {
				limit = { trait = anglachel }
				if = { 
					limit = { 
						is_ruler = yes
						current_heir = { 
							is_alive = yes  
							NOT = { trait = anglachel } 
						}
					}
					current_heir = {
						add_trait = anglachel 
					}
					remove_trait = anglachel
				}
			}	
	
		FROM = {
			clr_character_flag = abdicate
		}
		
		clr_character_flag = wants_to_abdicate
		
		abdicate_to = FROM
		
		set_character_flag = abdicated
		
		any_claim = {
			remove_claim = ROOT
		}
	}
